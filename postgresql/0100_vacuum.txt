-- Production rule: Alert on transactions open > 5 min, inactive replication slots, and backend_xmin age > 100M XIDs.

-- who is holding the xmin horizon back?
SELECT pid, usename, state, backend_xmin,
       now() - xact_start AS tx_age, query
FROM pg_stat_activity
WHERE backend_xmin IS NOT NULL
ORDER BY backend_xmin ASC
LIMIT 10;

-- replication slots
SELECT slot_name, active, age(xmin) AS xmin_age
FROM pg_replication_slots
ORDER BY xmin_age DESC;
